import usb.core
import usb.util

# Tenta encontrar o Arduino (Vendor ID e Product ID comuns, ajuste se necessário)
dev = usb.core.find(idVendor=0x2341, idProduct=0x0043)

if dev is None:
    print("Arduino não encontrado. Verifique a conexão OTG.")
else:
    print("Arduino encontrado!")

    try:
        dev.set_configuration()
        print("Configuração estabelecida. Pronto para comunicação!")
    except usb.core.USBError as e:
        print(f"Erro de permissão ou acesso: {e}")
